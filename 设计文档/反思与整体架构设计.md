## 一、模块定位与目标

### 1.1 为什么需要反思机制

没有反思的 AI 会重复犯同样的错误：

| 问题 | 表现 | 反思机制的作用 |
|------|------|----------------|
| 重复失败 | 挖矿失败后继续挖 | 分析原因，尝试不同地点 |
| 死板执行 | 不懂得变通 | 失败后调整策略 |
| 不知道问题 | 不知道为什么失败 | 显式分析错误原因 |
| 不知道成功 | 不知道为什么会成功 | 固化成功经验 |

### 1.2 反思机制的目标

```
核心功能：
1. 动作失败时分析原因
2. 生成替代方案
3. 将教训存入知识库
4. 避免重复犯错
```

### 2.2 critic.py - 反思器主类

```python
"""
ai/critic/critic.py
反思器主类
"""

from typing import Dict, List, Any, Optional
from dataclasses import dataclass
from enum import Enum
import json

from .prompts import REFLECT_PROMPT, FAILURE_ANALYSIS_PROMPT, SUCCESS_LEARNING_PROMPT
from .failure_analyzer import FailureAnalyzer
from .strategy import StrategyManager


class ReflectionType(Enum):
    """反思类型"""
    FAILURE = "failure"
    SUCCESS = "success"
    CONSECUTIVE_FAILURE = "consecutive_failure"
    GOAL_STALL = "goal_stall"
    UNEXPECTED_SUCCESS = "unexpected_success"


@dataclass
class ReflectionResult:
    """反思结果"""
    type: ReflectionType
    success: bool
    reason: str
    lesson: str
    suggested_action: Optional[Dict] = None
    strategy_change: Optional[str] = None
    knowledge_entry: Optional[Dict] = None


class Critic:
    """
    反思器
    
    职责：
    1. 分析动作执行结果
    2. 生成反思
    3. 调整策略
    4. 更新知识库
    """
    
    def __init__(self, llm_client, knowledge_base=None):
        self.llm = llm_client
        self.knowledge = knowledge_base
        self.failure_analyzer = FailureAnalyzer()
        self.strategy = StrategyManager()
        self._reflection_history = []
    
    def reflect(
        self,
        action: Dict[str, Any],
        result: Dict[str, Any],
        context: Dict[str, Any]
    ) -> ReflectionResult:
        """
        执行反思
        
        Args:
            action: 执行的动作
            result: 执行结果
            context: 上下文（目标、状态等）
            
        Returns:
            反思结果
        """
        # 1. 判断反思类型
        reflection_type = self._determine_type(action, result)
        
        # 2. 生成反思
        if reflection_type == ReflectionType.FAILURE:
            return self._reflect_on_failure(action, result, context)
        elif reflection_type == ReflectionType.SUCCESS:
            return self._reflect_on_success(action, result, context)
        elif reflection_type == ReflectionType.CONSECUTIVE_FAILURE:
            return self._reflect_on_consecutive_failure(action, result, context)
        else:
            return self._reflect_on_success(action, result, context)
    
    def _determine_type(self, action: Dict, result: Dict) -> ReflectionType:
        """判断反思类型"""
        if result.get('success'):
            # 检查是否意外成功
            if result.get('unexpected'):
                return ReflectionType.UNEXPECTED_SUCCESS
            return ReflectionType.SUCCESS
        else:
            # 检查是否连续失败
            recent_failures = sum(
                1 for h in self._reflection_history[-5:]
                if not h.success
            )
            if recent_failures >= 3:
                return ReflectionType.CONSECUTIVE_FAILURE
            return ReflectionType.FAILURE
    
    def _reflect_on_failure(
        self,
        action: Dict,
        result: Dict,
        context: Dict
    ) -> ReflectionResult:
        """反思失败"""
        error = result.get('error', 'Unknown error')
        
        # 1. 使用规则分析器快速判断
        quick_analysis = self.failure_analyzer.analyze(action, error)
        
        if quick_analysis:
            # 有规则匹配的快速分析
            lesson = quick_analysis['lesson']
            suggested = quick_analysis.get('suggested_action')
            
            # 存入知识库
            if self.knowledge and quick_analysis.get('add_to_knowledge'):
                self.knowledge.add(
                    situation=f"执行 {action.get('type')} 时遇到 {error}",
                    lesson=lesson,
                    category='failure'
                )
            
            # 记录历史
            reflection = ReflectionResult(
                type=ReflectionType.FAILURE,
                success=False,
                reason=error,
                lesson=lesson,
                suggested_action=suggested,
                knowledge_entry={
                    'situation': f"执行 {action.get('type')} 时遇到 {error}",
                    'lesson': lesson
                }
            )
            self._reflection_history.append(reflection)
            return reflection
        
        # 2. 没有规则匹配，使用 LLM 分析
        prompt = FAILURE_ANALYSIS_PROMPT.format(
            action_type=action.get('type'),
            action_data=json.dumps(action.get('data', {}), ensure_ascii=False),
            error=error,
            current_goal=context.get('current_goal', 'Unknown'),
            recent_actions=context.get('recent_history', 'None')
        )
        
        try:
            response = self.llm.chat.completions.create(
                model="deepseek-v3.2",
                messages=[{"role": "user", "content": prompt}],
                temperature=0.1,
                max_tokens=500
            )
            
            content = response.choices[0].message.content.strip()
            
            # 解析 JSON
            if content.startswith('```'):
                content = '\n'.join(content.split('\n')[1:])
                if content.endswith('```'):
                    content = content[:-3]
            
            analysis = json.loads(content.strip())
            
            # 更新策略
            if analysis.get('strategy_change'):
                self.strategy.adjust(analysis['strategy_change'])
            
            # 存入知识库
            if self.knowledge and analysis.get('lesson'):
                self.knowledge.add(
                    situation=analysis.get('situation', error),
                    lesson=analysis.get('lesson', ''),
                    category='failure'
                )
            
            reflection = ReflectionResult(
                type=ReflectionType.FAILURE,
                success=False,
                reason=error,
                lesson=analysis.get('lesson', ''),
                suggested_action=analysis.get('suggested_action'),
                strategy_change=analysis.get('strategy_change'),
                knowledge_entry={
                    'situation': analysis.get('situation', error),
                    'lesson': analysis.get('lesson', '')
                }
            )
            self._reflection_history.append(reflection)
            return reflection
            
        except Exception as e:
            print(f"⚠️ 反思失败: {e}")
            return ReflectionResult(
                type=ReflectionType.FAILURE,
                success=False,
                reason=error,
                lesson=f"遇到错误: {error}",
                suggested_action=None
            )
    
    def _reflect_on_success(
        self,
        action: Dict,
        result: Dict,
        context: Dict
    ) -> ReflectionResult:
        """反思成功"""
        # 成功时的反思主要用于提炼经验
        
        # 1. 检查是否值得学习
        if not result.get('learning_worthy'):
            return ReflectionResult(
                type=ReflectionType.SUCCESS,
                success=True,
                reason="Success",
                lesson="",
                suggested_action=None
            )
        
        # 2. 生成学习要点
        prompt = SUCCESS_LEARNING_PROMPT.format(
            action_type=action.get('type'),
            result=json.dumps(result, ensure_ascii=False),
            current_goal=context.get('current_goal', 'Unknown')
        )
        
        try:
            response = self.llm.chat.completions.create(
                model="deepseek-v3.2",
                messages=[{"role": "user", "content": prompt}],
                temperature=0.1,
                max_tokens=300
            )
            
            content = response.choices[0].message.content.strip()
            
            # 解析
            if '```' in content:
                lines = content.split('\n')
                content = '\n'.join(lines[1:] if lines[0] == '```' else lines)
                if content.endswith('```'):
                    content = content[:-3]
            
            lesson = content.strip()
            
            # 存入知识库
            if self.knowledge and lesson:
                self.knowledge.add(
                    situation=f"{action.get('type')} 成功完成",
                    lesson=lesson,
                    category='success'
                )
            
            reflection = ReflectionResult(
                type=ReflectionType.SUCCESS,
                success=True,
                reason="Success",
                lesson=lesson,
                suggested_action=None,
                knowledge_entry={
                    'situation': f"{action.get('type')} 成功完成",
                    'lesson': lesson
                }
            )
            self._reflection_history.append(reflection)
            return reflection
            
        except Exception as e:
            print(f"⚠️ 成功反思失败: {e}")
            return ReflectionResult(
                type=ReflectionType.SUCCESS,
                success=True,
                reason="Success",
                lesson="",
                suggested_action=None
            )
    
    def _reflect_on_consecutive_failure(
        self,
        action: Dict,
        result: Dict,
        context: Dict
    ) -> ReflectionResult:
        """反思连续失败"""
        # 连续失败时，需要更激进的策略调整
        
        # 1. 检查知识库是否有相关经验
        if self.knowledge:
            relevant = self.knowledge.retrieve_relevant(
                context=action.get('type'),
                limit=3
            )
            
            if relevant:
                lesson = relevant[0]['lesson']
                return ReflectionResult(
                    type=ReflectionType.CONSECUTIVE_FAILURE,
                    success=False,
                    reason="Consecutive failures",
                    lesson=f"连续失败！{lesson}",
                    suggested_action=None,
                    strategy_change="try_alternative_approach"
                )
        
        # 2. 建议放弃当前路径
        return ReflectionResult(
            type=ReflectionType.CONSECUTIVE_FAILURE,
            success=False,
            reason="Consecutive failures",
            lesson="连续失败，建议换个思路",
            suggested_action=None,
            strategy_change="switch_approach"
        )
    
    def get_advice(self, context: str) -> Optional[str]:
        """获取相关建议"""
        if not self.knowledge:
            return None
        
        relevant = self.knowledge.retrieve_relevant(context=context, limit=2)
        if relevant:
            return relevant[0]['lesson']
        return None
    
    def get_recent_lessons(self, n: int = 5) -> List[str]:
        """获取最近的教训"""
        lessons = []
        for h in reversed(self._reflection_history[-n*2:]):
            if h.lesson:
                lessons.append(h.lesson)
                if len(lessons) >= n:
                    break
        return lessons
```

"""
ai/critic/failure_analyzer.py
基于规则的失败分析器

对于常见的错误，提供快速分析和解决方案
"""

from typing import Dict, List, Optional, Tuple


class FailureAnalyzer:
    """
    规则匹配失败分析器
    
    预先定义常见错误的分析和解决方案
    """
    
    # 规则列表：匹配模式 -> (教训, 建议动作, 是否存入知识库)
    RULES: List[Tuple[str, str, Dict, bool]] = [
        # 导航相关
        (
            ".*path.*not.*found.*|.*no.*path.*",
            "无法到达目标位置，可能被障碍物阻挡或目标在水下/空中",
            {"type": "navigate", "data": {"mode": "moveTo", "adjust": True}},
            True
        ),
        (
            ".*too.*far.*|.*distance.*",
            "目标太远，超出寻路范围",
            None,
            True
        ),
        (
            ".*goal.*timeout.*|.*timeout.*",
            "移动超时，可能是路径太长或遇到障碍",
            None,
            False
        ),
        
        # 挖掘相关
        (
            ".*no.*tool.*|.*cannot.*harvest.*",
            "没有合适的工具挖掘这个方块",
            {"type": "equip", "data": {"keyword": ["pickaxe"]}},
            True
        ),
        (
            ".*cannot.*dig.*|.*block.*protected.*",
            "无法挖掘这个方块，可能是被保护或有实体遮挡",
            {"type": "navigate", "data": {"mode": "moveTo", "adjust": True}},
            True
        ),
        (
            ".*no.*block.*|.*cannot.*find.*",
            "附近没有找到目标方块",
            {"type": "navigate", "data": {"mode": "explore"}},
            True
        ),
        (
            ".*in.*lava.*|.*lava.*",
            "掉进岩浆！挖矿时要注意安全",
            {"type": "navigate", "data": {"mode": "retreat"}},
            True
        ),
        
        # 战斗相关
        (
            ".*no.*weapon.*|.*cannot.*attack.*",
            "没有武器可以攻击",
            {"type": "equip", "data": {"keyword": ["sword"]}},
            True
        ),
        (
            ".*no.*target.*|.*target.*not.*found.*",
            "没有找到攻击目标",
            None,
            False
        ),
        
        # 物品相关
        (
            ".*no.*item.*|.*not.*in.*inventory.*|.*missing.*",
            "背包中没有所需物品",
            {"type": "navigate", "data": {"mode": "find", "item": "relevant"}},
            True
        ),
        (
            ".*inventory.*full.*|.*full.*",
            "背包已满，无法拾取物品",
            {"type": "itemManage", "data": {"mode": "organize"}},
            True
        ),
        
        # 合成相关
        (
            ".*no.*recipe.*|.*cannot.*craft.*|.*missing.*material.*",
            "没有足够的材料合成",
            {"type": "navigate", "data": {"mode": "gather", "material": "relevant"}},
            True
        ),
        (
            ".*no.*crafting.*table.*|.*table.*not.*found.*",
            "附近没有工作台",
            {"type": "placeBlock", "data": {"item": "crafting_table"}},
            True
        ),
        
        # 下界相关
        (
            ".*ghast.*|.*fireball.*",
            "被恶魂火球攻击！远离开阔区域",
            None,
            True
        ),
        (
            ".*piglin.*|.*zombified.*piglin.*",
            "被猪灵攻击！在猪灵要塞要戴金护甲或隐藏",
            {"type": "equip", "data": {"slot": "head", "keyword": ["golden"]}},
            True
        ),
        
        # 通用
        (
            ".*death.*|.*died.*",
            "Bot 死亡，检查周围环境避免再次死亡",
            {"type": "stopAll", "data": {}},
            False
        ),
        (
            ".*kicked.*|.*disconnected.*",
            "被服务器踢出，可能是网络问题或违规操作",
            None,
            False
        ),
    ]
    
    def __init__(self):
        import re
        self.patterns = [
            (re.compile(pattern, re.IGNORECASE), lesson, suggested, add_kb)
            for pattern, lesson, suggested, add_kb in self.RULES
        ]
    
    def analyze(self, action: Dict, error: str) -> Optional[Dict]:
        """
        分析错误
        
        Args:
            action: 执行的动作用于提取上下文
            error: 错误信息
            
        Returns:
            分析结果 {"lesson": ..., "suggested_action": ...}
            或 None（没有匹配）
        """
        for pattern, lesson, suggested, add_kb in self.patterns:
            if pattern.search(error):
                result = {
                    'lesson': lesson,
                    'suggested_action': self._fill_suggestion(suggested, action),
                    'add_to_knowledge': add_kb
                }
                return result
        
        return None
    
    def _fill_suggestion(self, suggested: Optional[Dict], action: Dict) -> Optional[Dict]:
        """填充建议动作的参数"""
        if not suggested:
            return None
        
        result = dict(suggested)
        
        # 从原动作提取相关参数
        action_type = action.get('type')
        action_data = action.get('data', {})
        
        if result.get('data', {}).get('keyword') == ['relevant']:
            result['data']['keyword'] = self._infer_keywords(action_type, action_data)
        
        if result.get('data', {}).get('item') == 'relevant':
            result['data']['item'] = action_data.get('targetBlock') or action_data.get('item', 'unknown')
        
        return result
    
    def _infer_keywords(self, action_type: str, action_data: Dict) -> List[str]:
        """从动作推断关键词"""
        if action_type == 'dig':
            block = action_data.get('targetBlock', '')
            if 'diamond' in block:
                return ['pickaxe']
            elif 'iron' in block:
                return ['pickaxe']
            elif 'stone' in block:
                return ['pickaxe']
            return ['tool']
        
        if action_type == 'attack':
            return ['sword', 'axe']
        
        return ['item']
```

### 2.5 strategy.py - 策略管理器

```python
"""
ai/critic/strategy.py
策略管理器

管理 AI 的行为策略，如探索方式、战斗风格等
"""

from typing import Dict, List, Optional
from dataclasses import dataclass
from enum import Enum


class Strategy(Enum):
    """策略类型"""
    CAUTIOUS = "cautious"      # 谨慎
    AGGRESSIVE = "aggressive"  # 激进
    EXPLORER = "explorer"      # 探索优先
    EFFICIENT = "efficient"     # 效率优先
    DEFAULT = "default"         # 默认


@dataclass
class StrategyProfile:
    """策略配置"""
    name: str
    explore_before_action: bool = True
    check_inventory_before: bool = True
    prefer_safe_paths: bool = True
    retreat_on_low_health: bool = True
    min_health_threshold: float = 8.0
    min_food_threshold: float = 6.0
    max_attempts_before_giveup: int = 3


class StrategyManager:
    """
    策略管理器
    
    根据情况动态调整 AI 行为策略
    """
    
    # 预定义策略
    PROFILES: Dict[Strategy, StrategyProfile] = {
        Strategy.DEFAULT: StrategyProfile(
            name="默认",
            explore_before_action=True,
            check_inventory_before=True,
            prefer_safe_paths=True,
            retreat_on_low_health=True,
            min_health_threshold=8.0,
            min_food_threshold=6.0,
            max_attempts_before_giveup=3,
        ),
        Strategy.CAUTIOUS: StrategyProfile(
            name="谨慎",
            explore_before_action=True,
            check_inventory_before=True,
            prefer_safe_paths=True,
            retreat_on_low_health=True,
            min_health_threshold=12.0,
            min_food_threshold=10.0,
            max_attempts_before_giveup=2,
        ),
        Strategy.AGGRESSIVE: StrategyProfile(
            name="激进",
            explore_before_action=False,
            check_inventory_before=False,
            prefer_safe_paths=False,
            retreat_on_low_health=False,
            min_health_threshold=4.0,
            min_food_threshold=4.0,
            max_attempts_before_giveup=5,
        ),
        Strategy.EXPLORER: StrategyProfile(
            name="探索优先",
            explore_before_action=True,
            check_inventory_before=True,
            prefer_safe_paths=False,
            retreat_on_low_health=True,
            min_health_threshold=8.0,
            min_food_threshold=8.0,
            max_attempts_before_giveup=3,
        ),
        Strategy.EFFICIENT: StrategyProfile(
            name="效率优先",
            explore_before_action=False,
            check_inventory_before=False,
            prefer_safe_paths=True,
            retreat_on_low_health=True,
            min_health_threshold=6.0,
            min_food_threshold=4.0,
            max_attempts_before_giveup=2,
        ),
    }
    
    def __init__(self):
        self.current_strategy = Strategy.DEFAULT
        self.history: List[str] = []
    
    def get_profile(self) -> StrategyProfile:
        """获取当前策略配置"""
        return self.PROFILES[self.current_strategy]
    
    def adjust(self, change: str):
        """
        调整策略
        
        Args:
            change: 策略调整指令
        """
        change = change.lower()
        
        if 'cautious' in change or 'careful' in change or 'safe' in change:
            self._switch_to(Strategy.CAUTIOUS)
        elif 'aggressive' in change or 'rush' in change:
            self._switch_to(Strategy.AGGRESSIVE)
        elif 'explore' in change or 'scout' in change:
            self._switch_to(Strategy.EXPLORER)
        elif 'efficient' in change or 'fast' in change:
            self._switch_to(Strategy.EFFICIENT)
        elif 'default' in change:
            self._switch_to(Strategy.DEFAULT)
    
    def _switch_to(self, strategy: Strategy):
        """切换策略"""
        if strategy != self.current_strategy:
            self.current_strategy = strategy
            profile = self.PROFILES[strategy]
            self.history.append(f"Switched to {profile.name}")
            print(f"📊 策略调整: {profile.name}")
    
    def should_retry(self, attempts: int) -> bool:
        """判断是否应该重试"""
        profile = self.get_profile()
        return attempts < profile.max_attempts_before_giveup
    
    def should_retreat(self, health: float, food: float) -> bool:
        """判断是否应该撤退"""
        profile = self.get_profile()
        return (health < profile.min_health_threshold or 
                food < profile.min_food_threshold)
    
    def get_combat_strategy(self) -> str:
        """获取战斗策略"""
        if self.current_strategy == Strategy.AGGRESSIVE:
            return "rush"  # 直接冲上去打
        elif self.current_strategy == Strategy.CAUTIOUS:
            return "kite"  # 放风筝
        else:
            return "balanced"  # 平衡
```
