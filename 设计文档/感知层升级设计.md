## 一、模块定位与目标

### 1.1 现有问题

当前 `js/perception/scan.js` 存在以下根本性问题：

| 问题 | 现状 | 影响 |
|------|------|------|
| 数据量过大 | 每次返回数千个原始方块数据 | LLM 无法高效处理，token 浪费严重 |
| 语义缺失 | 只返回 `{name, x, y, z}` | Bot 不知道"自己在干什么" |
| 缺少高层感知 | 无生物群系/装备等级/威胁评估 | 无法做出战略性决策 |
| 原始数据为主 | 直接遍历 60³ 方块 | 性能开销大，信息密度低 |

### 1.2 升级目标

```
感知层 = 数据采集 + 语义提炼 + 格式转换
```

最终目标是：**把原始游戏状态，提炼成 LLM 可以高效理解的语义摘要**。

## 二、感知模块架构

### 2.1 整体结构

```
js/perception/
├── scan.js              # [保留] 底层方块/实体扫描（内部使用）
├── environment.js        # [新增] 环境语义感知
├── threat.js             # [新增] 威胁评估
├── resources.js          # [新增] 资源与装备感知
├── progress.js           # [新增] 游戏进度感知
├── worldMap.js           # [新增] 地图记忆感知
├── aggregator.js         # [新增] 感知聚合器（对外统一接口）
└── index.js             # [新增] 模块导出
```

### 2.2 感知层级划分

```
Level 1 - 原始数据（内部使用）
  └── mineflayer API 原始调用

Level 2 - 局部感知（每个子模块）
  └── environment.js / threat.js / resources.js

Level 3 - 聚合感知（aggregator.js）
  └── 所有子模块的结果合并 + 格式化

Level 4 - LLM 友好格式（server.py 转换）
  └── 转为自然语言描述或精简 JSON
```

### 3.1 environment.js - 环境语义感知

**职责**：感知当前位置的地形、环境、时间等高层语义信息。

```javascript
/**
 * 环境语义感知
 * @description 将原始位置信息提炼为 LLM 容易理解的环境描述
 */

/**
 * @typedef {Object} EnvironmentState
 * @property {Object} position - 当前位置 {x, y, z}
 * @property {string} dimension - 维度 'overworld' | 'the_nether' | 'the_end'
 * @property {string} biome - 生物群系名称（英文）
 * @property {string} biomeZh - 生物群系中文名
 * @property {boolean} isUnderground - 是否在地下（y < 50 或被天花板覆盖）
 * @property {boolean} isOnSurface - 是否在地表
 * @property {number} lightLevel - 光照等级 0-15
 * @property {string} timeOfDay - 时间描述 'dawn' | 'day' | 'dusk' | 'night'
 * @property {number} dayTick - 游戏刻时间 0-24000
 * @property {boolean} isRaining - 是否下雨
 * @property {Object} terrain - 地形特征
 */

/**
 * 地形特征检测
 * @param {Object} bot - mineflayer bot 实例
 * @returns {Object} 地形特征对象
 */
function detectTerrain(bot) {
    const pos = bot.entity.position
    const mcData = require('minecraft-data')(bot.version)
    
    // 检测脚下和头顶
    const below = bot.blockAt(pos.offset(0, -1, 0))
    const above = bot.blockAt(pos.offset(0, 1, 0))
    
    // 检测附近是否有特定方块
    const nearbyWater = bot.findBlock({
        matching: b => b.name === 'water',
        maxDistance: 16
    })
    const nearbyLava = bot.findBlock({
        matching: b => b.name === 'lava',
        maxDistance: 16
    })
    const nearbyTree = bot.findBlock({
        matching: b => b.name.includes('log'),
        maxDistance: 16
    })
    
    // 简单的洞穴检测：如果脚下是石头且上方封闭，可能在洞穴中
    let isInCave = false
    if (below && below.name.includes('stone')) {
        let ceilingCount = 0
        for (let y = pos.y + 1; y <= pos.y + 5; y++) {
            const block = bot.blockAt(new Vec3(Math.floor(pos.x), y, Math.floor(pos.z)))
            if (block && block.name !== 'air') ceilingCount++
        }
        isInCave = ceilingCount >= 4
    }
    
    return {
        belowBlock: below?.name || 'air',
        aboveBlock: above?.name || 'air',
        hasWater: !!nearbyWater,
        hasLava: !!nearbyLava,
        hasTree: !!nearbyTree,
        isInCave,
        caveDepth: isInCave ? pos.y : null,
    }
}

/**
 * 时间描述转换
 * @param {number} tick - 游戏刻 0-24000
 * @returns {Object} 时间信息
 */
function parseTimeOfDay(tick) {
    if (tick < 1000) return { timeOfDay: 'dawn', dayPhase: '日出' }
    if (tick < 13000) return { timeOfDay: 'day', dayPhase: '白天' }
    if (tick < 14000) return { timeOfDay: 'dusk', dayPhase: '黄昏' }
    return { timeOfDay: 'night', dayPhase: '夜晚' }
}

/**
 * 获取完整环境感知
 * @param {Object} bot - mineflayer bot 实例
 * @returns {EnvironmentState}
 */
function getEnvironmentState(bot) {
    const pos = bot.entity.position
    const block = bot.blockAt(pos)
    const biome = block?.biome?.name || 'unknown'
    
    return {
        position: {
            x: Math.floor(pos.x),
            y: Math.floor(pos.y),
            z: Math.floor(pos.z)
        },
        dimension: bot.game.dimension,
        biome,
        biomeZh: translateBiome(biome),
        isUnderground: pos.y < 50 || detectTerrain(bot).isInCave,
        isOnSurface: pos.y >= 50 && !detectTerrain(bot).isInCave,
        lightLevel: block?.light || 0,
        ...parseTimeOfDay(bot.time.timeOfDay),
        dayTick: bot.time.timeOfDay,
        isRaining: bot.isRaining,
        terrain: detectTerrain(bot),
    }
}

/**
 * 生物群系中英文对照（部分）
 */
const BIOME_TRANSLATIONS = {
    'plains': '平原',
    'forest': '森林',
    'birch_forest': '桦木森林',
    'dark_forest': '黑森林',
    'taiga': '针叶林',
    'desert': '沙漠',
    'savanna': '热带草原',
    'mountain': '山地',
    'jungle': '丛林',
    'swamp': '沼泽',
    'ice_plains': '冰原',
    'mushroom_island': '蘑菇岛',
    'beach': '海滩',
    'river': '河流',
    'ocean': '海洋',
    'deep_ocean': '深海',
}

function translateBiome(biome) {
    return BIOME_TRANSLATIONS[biome] || biome
}

module.exports = { getEnvironmentState, detectTerrain, parseTimeOfDay }
```

### 3.2 threat.js - 威胁评估

**职责**：评估当前环境的安全程度，包括敌对生物、血量、饱食度等。

```javascript
/**
 * 威胁评估模块
 * @description 感知周围危险程度，评估 Bot 生存状态
 */

/**
 * @typedef {Object} ThreatState
 * @property {number} hostileCount - 附近敌对生物数量
 * @property {number} nearestHostileDistance - 最近敌对生物距离
 * @property {Array<string>} hostileTypes - 敌对生物种类列表
 * @property {boolean} isInDanger - 是否处于危险状态
 * @property {string} dangerReason - 危险原因（如果危险）
 * @property {number} health - 当前血量
 * @property {number} food - 当前饱食度
 * @property {boolean} isHungry - 是否饥饿（饱食度 < 8）
 * @property {boolean} isLowHealth - 是否低血量（血量 < 8）
 * @property {number} armorRating - 护甲评级 0-4
 * @property {string} bestWeapon - 最佳武器名称
 * @property {number} weaponDamage - 最佳武器伤害
 */

// 敌对生物列表（来自 minecraft-data）
const HOSTILE_TYPES = [
    'zombie', 'skeleton', 'spider', 'creeper', 'enderman',
    'witch', 'pillager', 'vindicator', 'ravager', 'vex',
    'evoker', 'blaze', 'ghast', 'fireball', 'wither_skeleton',
    'stray', 'drowned', 'husk', 'phantom', 'shulker',
    'magma_cube', 'zombie_villager', 'piglin_brute', 'hoglin',
    'zoglin', 'warden'
]

/**
 * 查找最佳武器
 * @param {Object} bot - mineflayer bot 实例
 * @returns {Object|null} 武器信息 {name, damage}
 */
function findBestWeapon(bot) {
    const mcData = require('minecraft-data')(bot.version)
    const items = bot.inventory.items()
    
    // 武器伤害表
    const weaponDamage = {
        'diamond_sword': 7,
        'golden_sword': 4,
        'iron_sword': 6,
        'stone_sword': 5,
        'wooden_sword': 4,
        'diamond_axe': 9,
        'golden_axe': 3,
        'iron_axe': 5,
        'stone_axe': 4,
        'wooden_axe': 3,
    }
    
    let bestWeapon = null
    let bestDamage = 0
    
    for (const item of items) {
        const damage = weaponDamage[item.name]
        if (damage && damage > bestDamage) {
            bestDamage = damage
            bestWeapon = item.name
        }
    }
    
    return bestWeapon ? { name: bestWeapon, damage: bestDamage } : null
}

/**
 * 评估护甲等级
 * @param {Object} bot - mineflayer bot 实例
 * @returns {Object} 护甲信息
 */
function getArmorRating(bot) {
    const slots = ['head', 'torso', 'legs', 'feet']
    let armorCount = 0
    let armorSets = { leather: 0, iron: 0, diamond: 0, golden: 0, chainmail: 0 }
    
    for (const slot of slots) {
        try {
            const item = bot.inventory.slots[bot.entity.equipmentInventory[slot]]
            if (item && item.name.includes('helmet')) armorCount++
            if (item) {
                for (const type of Object.keys(armorSets)) {
                    if (item.name.includes(type)) armorSets[type]++
                }
            }
        } catch (e) {}
    }
    
    // 找出最佳护甲套装
    let bestSet = 'none'
    let bestCount = 0
    for (const [type, count] of Object.entries(armorSets)) {
        if (count > bestCount) {
            bestCount = count
            bestSet = type
        }
    }
    
    return {
        rating: armorCount,
        bestSet,
        setCount: bestCount,
    }
}

/**
 * 获取完整威胁评估
 * @param {Object} bot - mineflayer bot 实例
 * @param {number} range - 检测范围，默认 20 格
 * @returns {ThreatState}
 */
function getThreatState(bot, range = 20) {
    const pos = bot.entity.position
    
    // 收集附近敌对生物
    const hostiles = []
    for (const id in bot.entities) {
        const entity = bot.entities[id]
        if (entity === bot.entity) continue
        
        const dist = entity.position.distanceTo(pos)
        if (dist <= range && HOSTILE_TYPES.includes(entity.name)) {
            hostiles.push({
                name: entity.name,
                distance: dist,
                position: {
                    x: Math.floor(entity.position.x),
                    y: Math.floor(entity.position.y),
                    z: Math.floor(entity.position.z)
                }
            })
        }
    }
    
    // 排序距离
    hostiles.sort((a, b) => a.distance - b.distance)
    
    // 危险判断
    const nearestDist = hostiles.length > 0 ? hostiles[0].distance : 999
    const isInDanger = bot.health < 8 || 
                       (bot.food < 6) || 
                       (hostiles.length > 0 && nearestDist < 5) ||
                       (hostiles.length >= 3)
    
    let dangerReason = null
    if (bot.health < 8) dangerReason = `血量过低: ${bot.health}/20`
    else if (bot.food < 6) dangerReason = `饱食度过低: ${bot.food}/20`
    else if (hostiles.length >= 3) dangerReason = `多个敌对生物包围: ${hostiles.length}个`
    else if (nearestDist < 5) dangerReason = `敌对生物过于接近: ${hostiles[0]?.name} 距离${Math.floor(nearestDist)}格`
    
    const weapon = findBestWeapon(bot)
    const armor = getArmorRating(bot)
    
    return {
        hostileCount: hostiles.length,
        nearestHostileDistance: nearestDist,
        hostileTypes: [...new Set(hostiles.map(h => h.name))],
        nearestHostile: hostiles[0] || null,
        allHostiles: hostiles,
        isInDanger,
        dangerReason,
        
        health: bot.health,
        food: bot.food,
        isHungry: bot.food < 8,
        isLowHealth: bot.health < 8,
        isStarving: bot.food <= 2,  // 即将无法恢复生命
        
        armor,
        weapon,
    }
}

/**
 * 生成危险警告消息
 * @param {ThreatState} threat
 * @returns {string|null}
 */
function getDangerWarning(threat) {
    if (!threat.isInDanger) return null
    
    const warnings = []
    
    if (threat.isLowHealth) {
        warnings.push(`⚠️ 血量危险: ${threat.health}/20`)
    }
    if (threat.isHungry) {
        warnings.push(`⚠️ 饥饿: ${threat.food}/20，需要进食`)
    }
    if (threat.hostileCount > 0) {
        warnings.push(`⚠️ ${threat.hostileCount}个敌对生物，最近${threat.nearestHostile?.name}距离${Math.floor(threat.nearestHostileDistance)}格`)
    }
    
    return warnings.join(' | ')
}

module.exports = { getThreatState, getDangerWarning, findBestWeapon, getArmorRating }
```

```javascript
/**
 * 统计背包中某类物品数量
 */
function countItems(items, keywords) {
    return items
        .filter(i => keywords.some(k => i.name.includes(k)))
        .reduce((sum, i) => sum + i.count, 0)
}

/**
 * 获取最佳工具等级
 */
function getBestToolLevel(items, type) {
    let maxLevel = 0
    let bestTool = null
    
    for (const item of items) {
        const level = TOOL_HIERARCHY[item.name]
        if (level && level > maxLevel) {
            maxLevel = level
            bestTool = item.name
        }
    }
    
    return { level: maxLevel, tool: bestTool }
}

/**
 * 获取最佳护甲
 */
function getBestArmorSet(items) {
    const armorSlots = ['helmet', 'chestplate', 'leggings', 'boots']
    let totalRating = 0
    let setCount = 0
    let setType = 'none'
    
    for (const item of items) {
        const rating = ARMOR_HIERARCHY[item.name]
        if (rating) {
            totalRating += rating
            setCount++
            // 提取护甲类型
            for (const type of ['diamond', 'iron', 'golden', 'leather', 'chainmail', 'turtle']) {
                if (item.name.includes(type)) {
                    setType = type
                }
            }
        }
    }
    
    if (setCount === 0) return { type: 'none', pieces: 0, rating: 0 }
    
    return { type: setType, pieces: setCount, rating: Math.round(totalRating) }
}

    const foodDetails = foodItems.slice(0, 5).map(i => ({ name: i.name, count: i.count }))
    
    // 空槽位
    const emptySlots = 36 - items.length
    
    return {
        summary,
        bestPickaxe: pickaxeName,
        bestAxe: PICKAXE_NAMES[axe.level] || 'none',
        bestSword: bestWeapon,
        swordDamage: bestDamage,
        bestArmor: armor.type,
        armorPieces: armor.pieces,
        armorRating: armor.rating,
        
        tools: items.filter(i => TOOL_HIERARCHY[i.name]),
        combat: {
            sword: sword?.name,
            bow: !!bow,
            crossbow: !!crossbow,
            shield: !!shield,
            trident: !!trident,
            fishingRod: !!fishingRod,
        },
        
        foodCount,
        foodDetails,
        
        emptySlots,
        totalItems: items.length,
    }
}

/**

    const items = bot.inventory.items()
    return items.some(i => i.name.includes(itemName))
}

/**
```
```javascript
 * 检查里程碑是否完成
 */
function isMilestoneComplete(milestone, resourceState) {
    if (milestone.requires.length === 0) return false  // 需要外部判断的跳过
    
    return milestone.requires.every(req => {
        if (resourceState.summary[req] !== undefined) {
            return resourceState.summary[req] > 0
        }
        return false
    })
}

/**
 * 获取游戏进度状态
 * @param {Object} bot - mineflayer bot 实例
 * @param {Object} resourceState - 资源状态（来自 resources.js）
 * @returns {ProgressState}
 */
function getProgressState(bot, resourceState) {
    const completedMilestones = []
    const pendingMilestones = []
    
    for (const milestone of MILESTONES) {
        if (isMilestoneComplete(milestone, resourceState)) {
            completedMilestones.push(milestone)
        } else {
            pendingMilestones.push(milestone)
        }
    }
    
    // 确定当前阶段
    const phaseOrder = ['早期', '中期', '中后期', '后期', '最终', '通关']
    let currentPhase = '未知'
    
    for (let i = phaseOrder.length - 1; i >= 0; i--) {
        if (completedMilestones.some(m => m.phase === phaseOrder[i])) {
            currentPhase = phaseOrder[i]
            break
        }
    }
    
    // 下一个目标
    const nextMilestone = pendingMilestones[0] || null
    
    // 完成度（粗略估计）
    const completionPercent = Math.round((completedMilestones.length / MILESTONES.length) * 100)
    
    // 关键判断
    const hasNetherPortal = resourceState.summary.obsidian >= 10
    const hasBlazePowder = resourceState.summary.blazePowder >= 2
    const hasEnderEyes = resourceState.summary.enderEye >= 12
    const inNether = bot.game.dimension === 'the_nether'
    const inEnd = bot.game.dimension === 'the_end'
    
    return {
        milestones: {
            completed: completedMilestones.map(m => m.id),
            pending: pendingMilestones.map(m => m.id),
        },
        currentPhase,
        nextMilestone: nextMilestone ? {
            id: nextMilestone.id,
            name: nextMilestone.name,
            phase: nextMilestone.phase,
        } : null,
        completionPercent,
        
        // 关键节点状态
        keyStates: {
            hasNetherPortal,
            hasBlazePowder,
            hasEnderEyes,
            inNether,
            inEnd,
            hasIron: resourceState.summary.ironIngot > 0,
            hasDiamond: resourceState.summary.diamond > 0,
        },
        
        // 详细里程碑列表
        milestoneDetails: MILESTONES.map(m => ({
            ...m,
            complete: isMilestoneComplete(m, resourceState)
        }))
    }
}

module.exports = { getProgressState, MILESTONES }
```

```javascript
/**

// 重要地点类型
const LOCATION_TYPES = {

// 特殊方块识别
const STRUCTURE_BLOCKS = {
   村庄: ['oak_log', 'birch_log', 'spruce_log'],

/**
 * 地点记录类
 */
class LocationRecord {
    constructor(type, position, name = null, metadata = {}) {
        this.type = type
        this.position = {
            x: Math.floor(position.x),
            y: Math.floor(position.y),
            z: Math.floor(position.z)
        }
        this.name = name
        this.metadata = metadata
        this.firstFound = Date.now()
        this.lastVisited = Date.now()
        this.visits = 1
    }
    
    visit() {
        this.lastVisited = Date.now()
        this.visits++
    }
}

/**
 * 世界地图管理器
 */
class WorldMap {
    constructor() {
        this.locations = []
        this.exploredBounds = null
        this.loadFromStorage()
    }
    
    /**
     * 添加或更新地点
     */
    addLocation(type, position, name = null, metadata = {}) {
        // 检查是否已存在
        const existing = this.locations.find(loc => {
            const dx = Math.abs(loc.position.x - Math.floor(position.x))
            const dz = Math.abs(loc.position.z - Math.floor(position.z))
            return loc.type === type && dx < 10 && dz < 10
        })
        
        if (existing) {
            existing.visit()
            Object.assign(existing.metadata, metadata)
            return existing
        }
        
        const newLoc = new LocationRecord(type, position, name, metadata)
        this.locations.push(newLoc)
        this.saveToStorage()
        return newLoc
    }
    
    /**
     * 获取指定类型的最近地点
     */
    getNearestOfType(type, position, maxDistance = Infinity) {
        const filtered = this.locations.filter(loc => loc.type === type)
        if (filtered.length === 0) return null
        
        let nearest = null
        let minDist = Infinity
        
        for (const loc of filtered) {
            const dist = Math.sqrt(
                Math.pow(loc.position.x - position.x, 2) +
                Math.pow(loc.position.z - position.z, 2)
            )
            if (dist < minDist && dist <= maxDistance) {
                minDist = dist
                nearest = loc
            }
        }
        
        return nearest ? { ...nearest, distance: Math.floor(minDist) } : null
    }
    
    /**
     * 获取所有指定类型的地点
     */
    getAllOfType(type) {
        return this.locations.filter(loc => loc.type === type)
    }
    
    /**
     * 更新探索边界
     */
    updateExploredBounds(position) {
        const x = Math.floor(position.x)
        const z = Math.floor(position.z)
        
        if (!this.exploredBounds) {
            this.exploredBounds = { minX: x, maxX: x, minZ: z, maxZ: z }
        }
        
        this.exploredBounds.minX = Math.min(this.exploredBounds.minX, x)
        this.exploredBounds.maxX = Math.max(this.exploredBounds.maxX, x)
        this.exploredBounds.minZ = Math.min(this.exploredBounds.minZ, z)
        this.exploredBounds.maxZ = Math.max(this.exploredBounds.maxZ, z)
    }
    
    /**
     * 检测探索覆盖率
     */
    getExplorationCoverage() {
        if (!this.exploredBounds) return { explored: false, area: 0 }
        
        const width = this.exploredBounds.maxX - this.exploredBounds.minX
        const height = this.exploredBounds.maxZ - this.exploredBounds.minZ
        const area = Math.abs(width * height)
        
        return {
            explored: true,
            minX: this.exploredBounds.minX,
            maxX: this.exploredBounds.maxX,
            minZ: this.exploredBounds.minZ,
            maxZ: this.exploredBounds.maxZ,
            width: Math.abs(width),
            height: Math.abs(height),
            area: Math.abs(area),
            center: {
                x: Math.floor((this.exploredBounds.minX + this.exploredBounds.maxX) / 2),
                z: Math.floor((this.exploredBounds.minZ + this.exploredBounds.maxZ) / 2),
            }
        }
    }
    
    /**
     * 持久化到文件
     */
    saveToStorage() {
        const fs = require('fs')
        const path = require('path')
        const dataPath = path.join(__dirname, '../../memory/world_map.json')
        
        try {
            fs.writeFileSync(dataPath, JSON.stringify({
                locations: this.locations,
                exploredBounds: this.exploredBounds,
                savedAt: new Date().toISOString()
            }, null, 2))
        } catch (e) {
            console.error('Failed to save world map:', e)
        }
    }
    
    /**
     * 从文件加载
     */
    loadFromStorage() {
        const fs = require('fs')
        const path = require('path')
        const dataPath = path.join(__dirname, '../../memory/world_map.json')
        
        try {
            if (fs.existsSync(dataPath)) {
                const data = JSON.parse(fs.readFileSync(dataPath, 'utf8'))
                this.locations = data.locations || []
                this.exploredBounds = data.exploredBounds
                console.log(`Loaded ${this.locations.length} world map locations`)
            }
        } catch (e) {
            console.error('Failed to load world map:', e)
        }
    }
    
    /**
     * 获取地图摘要
     */
    getSummary(position) {
        return {
            totalLocations: this.locations.length,
            home: this.getNearestOfType(LOCATION_TYPES.HOME, position, 1000),
            mines: this.getAllOfType(LOCATION_TYPES.MINE).slice(0, 3),
            villages: this.getAllOfType(LOCATION_TYPES.VILLAGE),
            netherPortal: this.locations.find(l => l.type === LOCATION_TYPES.PORTAL && l.metadata.dimension === 'nether'),
            stronghold: this.locations.find(l => l.type === LOCATION_TYPES.STRONGHOLD),
            exploration: this.getExplorationCoverage(),
        }
    }
}
```

### 3.6 aggregator.js - 感知聚合器

```javascript
/**
 * 感知聚合器
 * @description 整合所有感知模块，对外提供统一的感知状态
 */

const { getEnvironmentState } = require('./environment')
const { getThreatState, getDangerWarning } = require('./threat')
const { getResourceState } = require('./resources')
const { getProgressState } = require('./progress')
const { getWorldMap, LOCATION_TYPES } = require('./worldMap')

/**
 * @typedef {Object} AggregatedPerception
 * @property {Object} environment - 环境感知
 * @property {Object} threat - 威胁评估
 * @property {Object} resources - 资源感知
 * @property {Object} progress - 游戏进度
 * @property {Object} worldMap - 地图摘要
 * @property {string} dangerWarning - 危险警告（如有）
 * @property {string} summary - LLM 友好的自然语言摘要
 * @property {number} timestamp - 采集时间戳
 */

/**
 * 完整感知采集
 * @param {Object} bot - mineflayer bot 实例
 * @returns {AggregatedPerception}
 */
function getFullPerception(bot) {
    // 1. 并行采集各维度感知（性能优化）
    const environment = getEnvironmentState(bot)
    const threat = getThreatState(bot)
    const resources = getResourceState(bot)
    
    // 2. 更新地图记忆
    const worldMap = getWorldMap()
    worldMap.updateExploredBounds(bot.entity.position)
    
    // 3. 检测并记录重要地点
    detectAndRecordLocations(bot, worldMap)
    
    // 4. 游戏进度
    const progress = getProgressState(bot, resources)
    
    // 5. 危险警告
    const dangerWarning = getDangerWarning(threat)
    
    // 6. 生成自然语言摘要
    const summary = generatePerceptionSummary(environment, threat, resources, progress, worldMap)
    
    return {
        environment,
        threat,
        resources,
        progress,
        worldMap: worldMap.getSummary(bot.entity.position),
        dangerWarning,
        summary,
        timestamp: Date.now(),
    }
}

/**
 * 检测并记录重要地点
 */
function detectAndRecordLocations(bot, worldMap) {
    const pos = bot.entity.position
    const block = bot.blockAt(pos)
    
    // 检测箱子（可能是基地）
    const nearbyChest = bot.findBlock({
        matching: b => b.name === 'chest',
        maxDistance: 3
    })
    if (nearbyChest && !worldMap.getNearestOfType(LOCATION_TYPES.HOME, pos, 5)) {
        worldMap.addLocation(LOCATION_TYPES.HOME, nearbyChest.position, '自动记录基地')
    }
    
    // 检测下界传送门
    const nearbyPortal = bot.findBlock({
        matching: b => b.name === 'nether_portal' || b.name === 'portal',
        maxDistance: 3
    })
    if (nearbyPortal) {
        const existing = worldMap.locations.find(l => 
            l.type === LOCATION_TYPES.PORTAL && 
            Math.abs(l.position.x - nearbyPortal.position.x) < 5
        )
        if (!existing) {
            worldMap.addLocation(LOCATION_TYPES.PORTAL, nearbyPortal.position, '下界传送门', {
                dimension: bot.game.dimension
            })
        }
    }
    
    // 检测工作台
    const nearbyTable = bot.findBlock({
        matching: b => b.name === 'crafting_table',
        maxDistance: 3
    })
    if (nearbyTable) {
        const existing = worldMap.locations.find(l => 
            l.type === 'crafting_station' && 
            Math.abs(l.position.x - nearbyTable.position.x) < 5
        )
        if (!existing) {
            worldMap.addLocation('crafting_station', nearbyTable.position, '工作台')
        }
    }
}

/**
 * 生成 LLM 友好的自然语言摘要
 */
function generatePerceptionSummary(env, threat, res, progress, worldMap) {
    const lines = []
    
    // 位置与时间
    const timeStr = env.isDay ? '白天' : '夜晚'
    const rainStr = env.isRaining ? '（下雨）' : ''
    lines.push(`📍 位置: ${env.position.x}, ${env.position.y}, ${env.position.z} | ${env.biomeZh} | ${env.dimension} | ${timeStr}${rainStr}`)
    
    // 生存状态
    const healthIcon = threat.health > 15 ? '❤️' : threat.health > 8 ? '💔' : '💀'
    const foodIcon = threat.food > 15 ? '🍖' : threat.food > 8 ? '🍽️' : '🍴'
    lines.push(`${healthIcon} 血量: ${threat.health}/20 | ${foodIcon} 饱食: ${threat.food}/20 | 🛡️ 护甲: ${res.bestArmor}`)
    
    // 威胁
    if (threat.hostileCount > 0) {
        lines.push(`⚠️ 附近有 ${threat.hostileCount} 个 ${threat.hostileTypes.join(', ')}`)
    }
    
    // 关键装备
    const weaponStr = res.bestSword ? `⚔️ ${res.bestSword}(${res.swordDamage})` : '⚔️ 无武器'
    const pickStr = `⛏️ ${res.bestPickaxe}镐`
    lines.push(`${weaponStr} | ${pickStr}`)
    
    // 关键资源
    const keyResources = []
    if (res.summary.diamond > 0) keyResources.push(`💎 ${res.summary.diamond}`)
    if (res.summary.obsidian > 0) keyResources.push(`⬛ 黑曜石 ${res.summary.obsidian}`)
    if (res.summary.enderPearl > 0) keyResources.push(`🫐 珍珠 ${res.summary.enderPearl}`)
    if (res.summary.enderEye > 0) keyResources.push(`👁️ 末影眼 ${res.summary.enderEye}`)
    if (keyResources.length > 0) {
        lines.push(keyResources.join(' | '))
    }
    
    // 食物
    lines.push(`🍎 食物: ${res.foodCount} 个`)
    
    // 游戏进度
    lines.push(`📊 进度: ${progress.currentPhase} | 完成度 ~${progress.completionPercent}%`)
    if (progress.nextMilestone) {
        lines.push(`🎯 下一步: ${progress.nextMilestone.name}`)
    }
    
    // 危险警告
    if (threat.dangerReason) {
        lines.push(`🚨 ${threat.dangerReason}`)
    }
    
    return lines.join('\n')
}

/**
 * 生成精简版感知（用于高频轮询）

        isDay: bot.time.timeOfDay < 13000,
        hostileCount: Object.values(bot.entities).filter(e => 
            e !== bot.entity && e.type === 'hostile' && 
            e.position.distanceTo(pos) < 20
        ).length,
        timestamp: Date.now()
    }
}
```

## 七、性能优化策略

### 7.1 感知频率分级

| 感知类型 | 频率 | 数据量 | 使用场景 |
|---------|------|--------|---------|
| 快速感知 | 1Hz | ~50 bytes | 高频安全检查 |
| 完整感知 | 0.1Hz | ~2KB | 决策前上报 |
| 深度扫描 | 手动触发 | ~50KB | 探索新区域 |

### 7.2 缓存机制

```javascript
// 感知缓存，避免重复计算
let cachedPerception = null
let cacheTimestamp = 0
const CACHE_DURATION = 1000  // 1秒缓存

function getCachedPerception(bot) {
    if (Date.now() - cacheTimestamp < CACHE_DURATION) {
        return cachedPerception
    }
    
    cachedPerception = getFullPerception(bot)
    cacheTimestamp = Date.now()
    return cachedPerception
}
```

## 八、视觉感知（新增维度）

视觉感知作为独立的感知维度，与现有的文本感知互补，详见 `视觉系统设计.md`。

### 8.1 定位

```
文本感知：精确数据（坐标、类型、数量） → 高频、确定
视觉感知：结构和空间理解（建筑、地形） → 低频、辅助
```

### 8.2 感知合并后的最终结构

```javascript
// aggregator.js 增强版
function getFullPerception(bot) {
    // 1. 文本感知（原有逻辑）
    const text = collectTextPerception(bot)

    // 2. 视觉感知（按需触发）
    const vision = collectVisionPerception(bot)

    // 3. 合并
    return {
        ...text,
        vision: vision,       // 新增
        summary: generateSummary(text, vision)
    }
}
```

### 8.3 触发条件

视觉感知不在每次轮询中都执行，只在以下情况触发：

| 触发条件 | 说明 |
|---------|------|
| 首次到达新区域 | 大范围建图 |
| **LLM 主动请求** | 决策时视觉不足，返回 request_vision |
| 洞穴/室内环境 | 文本感知无法表达空间结构 |
| 目标涉及结构识别 | 找村庄、要塞、建筑等 |

### 8.4 六种视角

详见 `视觉系统设计.md`，各视角对应不同场景：

| 视角 | 适用场景 | 解决的问题 |
|------|---------|-----------|
| 第一人称 | 默认/快速检查 | 当前视野评估 |
| 环视全景 | 探索/战术决策 | 360° 环境评估 |
| 俯视图 45° | 建筑/地形分析 | 玩家建筑识别 |
| 洞穴切片 | 地下探索 | 洞穴结构理解 |
