## 一、模块定位与目标

### 1.1 为什么需要记忆系统

当前的 Bot 是**无状态**的：每次聊天指令独立处理，Bot 不知道自己之前做过什么。这导致：

| 问题 | 表现 | 影响 |
|------|------|------|
| 重复劳动 | 上次挖了 10 个钻石，下次又说"我没钻石" | 效率极低 |
| 不知道在哪 | 没有"家"的概念，飞走后找不到 | 无法规划返程 |
| 经验丢失 | 挖矿失败的原因，下次继续犯 | 学习能力为零 |
| 无法协作 | 不知道当前进度，无法承接玩家指令 | 智能化程度低 |

### 1.2 记忆系统的设计目标

```
短期记忆：保存最近 N 步的行动+结果，喂给 LLM 上下文
长期记忆：持久化世界状态（资源、位置、进度），跨 session 不丢失
知识记忆：保存学习到的经验/技巧，供决策时检索
```

## 二、记忆系统架构

### 2.1 三层记忆模型

```
┌─────────────────────────────────────────────────────────┐
│                    长期记忆（Long-term）                   │
│  世界状态 | 资源库存 | 探索地图 | 知识库                    │
│  ↓ 持久化到 disk（JSON/SQLite）                          │
├─────────────────────────────────────────────────────────┤
│                    短期记忆（Short-term）                 │
│  最近 20 步的行动序列（action + result + 反思）            │
│  ↓ 喂给 LLM 上下文窗口                                   │
├─────────────────────────────────────────────────────────┤
│                    工作记忆（Working Memory）              │
│  当前感知 + 当前目标 + 可用工具                            │
│  ↓ 当前决策循环内使用                                    │
└─────────────────────────────────────────────────────────┘
```

### 2.2 目录结构

```
ai/
├── memory/
│   ├── __init__.py
│   ├── memory.py          # 记忆主类
│   ├── short_term.py      # 短期记忆（行动序列）
│   ├── long_term.py       # 长期记忆（世界状态）
│   ├── knowledge.py       # 知识库（经验/技巧）
│   └── storage.py         # 持久化存储
memory/
├── world_state.json       # 世界状态（位置、资源、进度）
├── exploration_map.json   # 探索地图（已访问区块）
├── knowledge_base.json    # 知识库（经验教训）
└── action_history.json   # 行动历史日志
```

## 三、核心模块详细设计

### 3.1 memory.py - 记忆主类

**职责**：统一管理三层记忆，提供对外接口。

```python
"""
ai/memory/memory.py
记忆系统主类，统一管理短期、长期、知识记忆
"""

import json
import os
from datetime import datetime
from typing import Optional, List, Dict, Any
from pathlib import Path

from .short_term import ShortTermMemory
from .long_term import LongTermMemory
from .knowledge import KnowledgeBase


class Memory:
    """
    三层记忆管理器
    
    使用示例:
        memory = Memory()
        
        # 添加行动记录
        memory.add_action({
            'action': 'dig',
            'target': 'diamond_ore',
            'result': 'success',
            'count': 3,
            'position': {'x': 100, 'y': 40, 'z': -50}
        })
        
        # 获取上下文（供 LLM 使用）
        context = memory.get_context_for_llm(max_turns=10)
        
        # 更新世界状态
        memory.update_world_state({
            'position': {'x': 100, 'y': 40, 'z': -50},
            'resources': {'diamond': 5, 'iron_ingot': 20},
            'milestones': {'has_diamond_pickaxe': True}
        })
        
        # 保存经验
        memory.add_knowledge(
            situation='挖钻石时总是挖到岩浆',
            lesson='在 Y=11-12 层挖矿时要小心岩浆，先插火把确认安全'
        )
    """
    
    def __init__(self, memory_dir: str = '../memory'):
        self.memory_dir = Path(memory_dir)
        self.memory_dir.mkdir(exist_ok=True)
        
        # 初始化三层记忆
        self.short_term = ShortTermMemory(max_turns=20)
        self.long_term = LongTermMemory(self.memory_dir / 'world_state.json')
        self.knowledge = KnowledgeBase(self.memory_dir / 'knowledge_base.json')
        
        # 加载持久化数据
        self.long_term.load()
        self.knowledge.load()
        
    def add_action(self, action: Dict[str, Any], result: Dict[str, Any] = None,
                   reflection: str = None):
        """
        添加行动记录到短期记忆
        
        Args:
            action: 执行的动作 {'type': 'dig', 'target': 'diamond_ore', ...}
            result: 执行结果 {'success': True, 'count': 3, ...}
            reflection: 可选的反思笔记
        """
        entry = {
            'timestamp': datetime.now().isoformat(),
            'action': action,
            'result': result or {},
            'reflection': reflection
        }
        self.short_term.add(entry)
        
        # 如果成功，更新长期记忆中的资源状态
        if result and result.get('success'):
            self._update_resources_from_action(action, result)
    
    def _update_resources_from_action(self, action: Dict, result: Dict):
        """根据行动结果更新资源状态"""
        action_type = action.get('type')
        
        if action_type == 'dig':
            target = action.get('data', {}).get('targetBlock', 'unknown')
            count = result.get('count', 1)
            self.long_term.add_resources({target: count})
            
        elif action_type == 'itemManage' and action.get('data', {}).get('mode') == 'drop':
            item = action.get('data', {}).get('item', 'unknown')
            count = action.get('data', {}).get('count', 1)
            self.long_term.remove_resources({item: count})
    
    def get_context_for_llm(self, max_turns: int = 10,
                            include_knowledge: bool = True) -> str:
        """
        生成供 LLM 使用的上下文字符串
        
        Args:
            max_turns: 包含最近多少次行动
            include_knowledge: 是否包含知识库相关内容
            
        Returns:
            格式化的上下文字符串
        """
        parts = []
        
        # 1. 长期记忆（世界状态摘要）
        world_state = self.long_term.get_summary()
        if world_state:
            parts.append("【当前世界状态】")
            parts.append(world_state)
            parts.append("")
        
        # 2. 短期记忆（最近行动）
        recent_actions = self.short_term.get_recent(max_turns)
        if recent_actions:
            parts.append("【最近行动】")
            for entry in recent_actions:
                action = entry['action']
                result = entry.get('result', {})
                success = "✅" if result.get('success') else "❌"
                action_str = f"{success} {action.get('type', 'unknown')}"
                if action.get('data', {}).get('targetBlock'):
                    action_str += f" -> {action['data']['targetBlock']}"
                elif action.get('data', {}).get('position'):
                    pos = action['data']['position']
                    action_str += f" -> ({pos['x']}, {pos['y']}, {pos['z']})"
                parts.append(f"  - {action_str}")
            parts.append("")
        
        # 3. 相关知识（如果有）
        if include_knowledge:
            current_goal = self.long_term.get('current_goal')
            relevant_knowledge = self.knowledge.retrieve_relevant(
                context=current_goal,
                limit=3
            )
            if relevant_knowledge:
                parts.append("【相关经验】")
                for k in relevant_knowledge:
                    parts.append(f"  💡 {k['situation']}: {k['lesson']}")
                parts.append("")
        
        return "\n".join(parts)
    
    def update_world_state(self, state: Dict[str, Any]):
        """更新世界状态"""
        if 'position' in state:
            self.long_term.set_position(state['position'])
        if 'resources' in state:
            self.long_term.update_resources(state['resources'])
        if 'milestones' in state:
            self.long_term.update_milestones(state['milestones'])
        if 'dimension' in state:
            self.long_term.set('dimension', state['dimension'])
    
    def add_knowledge(self, situation: str, lesson: str, 
                      category: str = 'general'):
        """添加知识到知识库"""
        self.knowledge.add(situation, lesson, category)
    
    def add_failure_lessons(self, action: Dict, error: str, 
                           alternative_action: Dict = None):
        """
        记录失败教训
        
        Args:
            action: 失败的动作
            error: 错误原因
            alternative_action: 建议的替代动作
        """
        situation = f"执行 {action.get('type')} 时失败: {error}"
        lesson = f"失败原因: {error}"
        if alternative_action:
            lesson += f", 建议: 尝试 {alternative_action.get('type')}"
        
        self.knowledge.add(situation, lesson, category='failure')
    
    def get_exploration_hint(self) -> Optional[str]:
        """获取探索提示（引导去未探索区域）"""
        return self.long_term.get_exploration_hint()
    
    def save(self):
        """持久化所有记忆"""
        self.long_term.save()
        self.knowledge.save()
    
    def reset_short_term(self):
        """清空短期记忆（用于新任务开始）"""
        self.short_term = ShortTermMemory(max_turns=20)
```

### 3.2 short_term.py - 短期记忆

**职责**：管理最近 N 步的行动序列，用于 LLM 上下文。

```python
"""
ai/memory/short_term.py
短期记忆 - 维护最近的行动序列
"""

from typing import List, Dict, Any, Optional
from datetime import datetime, timedelta
from collections import deque


class ShortTermMemory:
    """
    短期记忆管理器
    
    特点:
    - 固定容量（FIFO）
    - 包含行动+结果+反思
    - 可序列化用于调试
    """
    
    def __init__(self, max_turns: int = 20):
        self.max_turns = max_turns
        self.history = deque(maxlen=max_turns)
        
    def add(self, entry: Dict[str, Any]):
        """添加行动记录"""
        if 'timestamp' not in entry:
            entry['timestamp'] = datetime.now().isoformat()
        self.history.append(entry)
    
    def get_recent(self, n: int = 10) -> List[Dict[str, Any]]:
        """获取最近 N 条记录"""
        items = list(self.history)[-n:]
        return items
    
    def get_formatted_history(self, n: int = 10) -> str:
        """获取格式化的历史记录"""
        items = self.get_recent(n)
        if not items:
            return "(无历史记录)"
        
        lines = []
        for i, entry in enumerate(items, 1):
            action = entry.get('action', {})
            result = entry.get('result', {})
            success = "✅" if result.get('success') else "❌"
            action_desc = self._format_action(action)
            lines.append(f"{i}. {success} {action_desc}")
        
        return "\n".join(lines)
    
    def _format_action(self, action: Dict) -> str:
        """格式化动作描述"""
        action_type = action.get('type', 'unknown')
        data = action.get('data', {})
        
        if action_type == 'navigate':
            pos = data.get('position', {})
            if pos:
                return f"移动到 ({pos.get('x')}, {pos.get('y')}, {pos.get('z')})"
            if data.get('mode') == 'follow':
                return f"跟随 {data.get('target')}"
            return "移动"
            
        elif action_type == 'dig':
            block = data.get('targetBlock', 'unknown')
            count = data.get('count', 1)
            return f"挖掘 {block} x{count}"
            
        elif action_type == 'attack':
            target = data.get('targetEntity', 'unknown')
            return f"攻击 {target}"
            
        elif action_type == 'useItem':
            mode = data.get('mode', 'unknown')
            item = data.get('item', '')
            return f"使用物品: {mode}" + (f" ({item})" if item else "")
            
        elif action_type == 'equip':
            item = data.get('item', 'unknown')
            slot = data.get('slot', '')
            return f"装备 {item} 到 {slot}"
            
        elif action_type == 'stopAll':
            return "停止所有动作"
            
        else:
            return action_type
    
    def get_success_rate(self, n: int = 20) -> float:
        """计算最近 N 次行动的成功率"""
        items = list(self.history)[-n:]
        if not items:
            return 0.0
        
        successes = sum(1 for item in items 
                       if item.get('result', {}).get('success', False))
        return successes / len(items)
    
    def get_last_action(self) -> Optional[Dict[str, Any]]:
        """获取最后一次行动"""
        return self.history[-1] if self.history else None
    
    def count_consecutive_failures(self) -> int:
        """计算连续失败次数"""
        count = 0
        for item in reversed(self.history):
            if item.get('result', {}).get('success', False):
                break
            count += 1
        return count
    
    def clear(self):
        """清空历史"""
        self.history.clear()
```

### 3.3 long_term.py - 长期记忆

**职责**：持久化管理世界状态（位置、资源、探索进度）。

```python
"""
ai/memory/long_term.py
长期记忆 - 持久化的世界状态
"""

import json
from pathlib import Path
from typing import Dict, Any, Optional, List
from datetime import datetime


class LongTermMemory:
    """
    长期记忆管理器
    
    持久化存储:
    - 当前位置和已探索区域
    - 累计资源数量
    - 游戏里程碑进度
    - 已知的重要地点
    """
    
    def __init__(self, storage_path: Path):
        self.storage_path = storage_path
        self.data: Dict[str, Any] = self._init_empty()
    
    def _init_empty(self) -> Dict[str, Any]:
        """初始化空数据结构"""
        return {
            'version': '1.0',
            'last_updated': datetime.now().isoformat(),
            
            # 位置信息
            'position': {
                'x': 0, 'y': 0, 'z': 0,
                'dimension': 'overworld'
            },
            
            # 出生点
            'spawn_point': None,
            
            # 累计资源
            'total_resources': {},
            
            # 资源里程碑
            'resource_milestones': {
                'first_diamond': False,
                'first_iron': False,
                'has_obsidian': False,
                'has_ender_eye': False,
            },
            
            # 游戏进度里程碑
            'progress_milestones': {
                'made_crafting_table': False,
                'made_furnace': False,
                'has_wooden_tools': False,
                'has_stone_tools': False,
                'has_iron_tools': False,
                'has_diamond_tools': False,
                'has_iron_armor': False,
                'has_diamond_armor': False,
                'entered_nether': False,
                'found_stronghold': False,
                'entered_end': False,
                'killed_dragon': False,
            },
            
            # 重要地点
            'locations': {
                'home': None,
                'mines': [],
                'villages': [],
                'nether_portal': None,
                'stronghold': None,
            },
            
            # 探索边界
            'exploration_bounds': {
                'min_x': None, 'max_x': None,
                'min_z': None, 'max_z': None,
            },
            
            # 当前目标
            'current_goal': None,
        }
    
    def load(self):
        """从文件加载"""
        if self.storage_path.exists():
            try:
                with open(self.storage_path, 'r', encoding='utf-8') as f:
                    loaded = json.load(f)
                    self.data = {**self._init_empty(), **loaded}
                    print(f"📚 已加载长期记忆，上次更新: {self.data.get('last_updated')}")
            except Exception as e:
                print(f"⚠️ 加载记忆失败: {e}，使用空白记忆")
                self.data = self._init_empty()
    
    def save(self):
        """保存到文件"""
        self.data['last_updated'] = datetime.now().isoformat()
        try:
            self.storage_path.parent.mkdir(parents=True, exist_ok=True)
            with open(self.storage_path, 'w', encoding='utf-8') as f:
                json.dump(self.data, f, indent=2, ensure_ascii=False)
            print(f"💾 长期记忆已保存")
        except Exception as e:
            print(f"❌ 保存记忆失败: {e}")
    
    def set(self, key: str, value: Any):
        self.data[key] = value
    
    def get(self, key: str, default: Any = None) -> Any:
        return self.data.get(key, default)
    
    def set_position(self, pos: Dict[str, int], dimension: str = None):
        """更新当前位置"""
        self.data['position'] = {
            'x': pos.get('x', 0),
            'y': pos.get('y', 0),
            'z': pos.get('z', 0),
            'dimension': dimension or self.data['position'].get('dimension', 'overworld')
        }
        self._update_exploration_bounds(pos)
    
    def _update_exploration_bounds(self, pos: Dict):
        """更新探索边界"""
        bounds = self.data['exploration_bounds']
        x, z = pos.get('x', 0), pos.get('z', 0)
        
        if bounds['min_x'] is None:
            bounds['min_x'] = bounds['max_x'] = x
            bounds['min_z'] = bounds['max_z'] = z
        else:
            bounds['min_x'] = min(bounds['min_x'], x)
            bounds['max_x'] = max(bounds['max_x'], x)
            bounds['min_z'] = min(bounds['min_z'], z)
            bounds['max_z'] = max(bounds['max_z'], z)
    
    def add_resources(self, resources: Dict[str, int]):
        """添加累计资源"""
        total = self.data['total_resources']
        for item, count in resources.items():
            total[item] = total.get(item, 0) + count
        self._check_resource_milestones()
    
    def remove_resources(self, resources: Dict[str, int]):
        """减少累计资源"""
        total = self.data['total_resources']
        for item, count in resources.items():
            total[item] = max(0, total.get(item, 0) - count)
    
    def _check_resource_milestones(self):
        """检查并更新资源里程碑"""
        total = self.data['total_resources']
        milestones = self.data['resource_milestones']
        
        if total.get('diamond', 0) > 0:
            milestones['first_diamond'] = True
        if total.get('iron_ingot', 0) > 0:
            milestones['first_iron'] = True
        if total.get('obsidian', 0) >= 10:
            milestones['has_obsidian'] = True
        if total.get('ender_eye', 0) >= 12:
            milestones['has_ender_eye'] = True
    
    def update_milestones(self, milestones: Dict[str, bool]):
        """更新进度里程碑"""
        self.data['progress_milestones'].update(milestones)
    
    def set_location(self, location_type: str, pos: Dict, name: str = None):
        """设置重要地点"""
        locations = self.data['locations']
        
        if location_type in ['home', 'nether_portal', 'stronghold']:
            locations[location_type] = {
                'position': pos,
                'name': name,
                'timestamp': datetime.now().isoformat()
            }
        elif location_type in ['mines', 'villages']:
            existing = next((i for i, loc in enumerate(locations[location_type])
                           if self._same_position(loc['position'], pos)), None)
            if existing is None:
                locations[location_type].append({
                    'position': pos,
                    'name': name,
                    'timestamp': datetime.now().isoformat()
                })
    
    def _same_position(self, pos1: Dict, pos2: Dict, threshold: int = 10) -> bool:
        return (abs(pos1['x'] - pos2['x']) < threshold and
                abs(pos1['z'] - pos2['z']) < threshold)
    
    def get_summary(self) -> str:
        """生成世界状态摘要（供 LLM 使用）"""
        pos = self.data['position']
        resources = self.data['total_resources']
        milestones = self.data['progress_milestones']
        bounds = self.data['exploration_bounds']
        
        lines = []
        
        dim_zh = {'overworld': '主世界', 'the_nether': '下界', 'the_end': '末地'}
        dim = dim_zh.get(pos.get('dimension', 'overworld'), pos.get('dimension'))
        lines.append(f"📍 当前位置: ({pos['x']}, {pos['y']}, {pos['z']}) | {dim}")
        
        if bounds['min_x'] is not None:
            width = abs(bounds['max_x'] - bounds['min_x'])
            height = abs(bounds['max_z'] - bounds['min_z'])
            lines.append(f"🗺️ 已探索: 宽{width}格 x 深{height}格")
        
        key_resources = []
        for item in ['diamond', 'iron_ingot', 'gold_ingot', 'obsidian', 
                     'ender_eye', 'ender_pearl', 'blaze_powder']:
            if resources.get(item, 0) > 0:
                key_resources.append(f"{item}: {resources[item]}")
        if key_resources:
            lines.append(f"💎 累计资源: {', '.join(key_resources)}")
        
        completed = [k for k, v in milestones.items() if v]
        if completed:
            lines.append(f"✅ 已完成里程碑: {len(completed)}/{len(milestones)}")
        
        home = self.data['locations'].get('home')
        if home:
            p = home['position']
            lines.append(f"🏠 基地: ({p['x']}, {p['y']}, {p['z']})")
        
        return "\n".join(lines)
    
    def get_exploration_hint(self) -> Optional[str]:
        """获取探索提示"""
        bounds = self.data['exploration_bounds']
        pos = self.data['position']
        
        if bounds['min_x'] is None:
            return "还没有开始探索，先收集木头"
        
        center_x = (bounds['min_x'] + bounds['max_x']) / 2
        center_z = (bounds['min_z'] + bounds['max_z']) / 2
        
        dx = pos['x'] - center_x
        dz = pos['z'] - center_z
        
        suggestions = []
        if abs(dx) > 200:
            suggestions.append("往" + ("东" if dx < 0 else "西"))
        if abs(dz) > 200:
            suggestions.append("往" + ("南" if dz < 0 else "北"))
        
        if suggestions:
            return "建议: " + "，".join(suggestions) + "探索新区域"
        
        return None
```

### 3.4 knowledge.py - 知识库

**职责**：存储和检索学习到的经验/技巧。

```python
"""
ai/memory/knowledge.py
知识库 - 存储和检索学习到的经验教训
"""

import json
from pathlib import Path
from typing import List, Dict, Any, Optional
from datetime import datetime


class KnowledgeBase:
    """
    知识库管理器
    
    存储类型:
    - 失败教训: 执行某动作时遇到的问题和解决方案
    - 成功经验: 有效的方法和技巧
    - 通用知识: Minecraft 游戏机制
    """
    
    def __init__(self, storage_path: Path):
        self.storage_path = storage_path
        self.entries: List[Dict[str, Any]] = []
        self.load()
    
    def load(self):
        """从文件加载"""
        if self.storage_path.exists():
            try:
                with open(self.storage_path, 'r', encoding='utf-8') as f:
                    data = json.load(f)
                    self.entries = data.get('entries', [])
                    print(f"📖 已加载 {len(self.entries)} 条知识")
            except Exception as e:
                print(f"⚠️ 加载知识库失败: {e}")
    
    def save(self):
        """保存到文件"""
        try:
            self.storage_path.parent.mkdir(parents=True, exist_ok=True)
            with open(self.storage_path, 'w', encoding='utf-8') as f:
                json.dump({
                    'version': '1.0',
                    'last_updated': datetime.now().isoformat(),
                    'entries': self.entries
                }, f, indent=2, ensure_ascii=False)
        except Exception as e:
            print(f"❌ 保存知识库失败: {e}")
    
    def add(self, situation: str, lesson: str, category: str = 'general'):
        """
        添加知识条目
        
        Args:
            situation: 问题场景描述
            lesson: 学到的教训/经验
            category: 分类 ['failure', 'success', 'general', 'tip']
        """
        # 检查是否已有类似条目
        for entry in self.entries:
            if self._is_similar(situation, entry.get('situation', '')):
                entry['lesson'] = lesson
                entry['reinforced'] = entry.get('reinforced', 0) + 1
                entry['last_reinforced'] = datetime.now().isoformat()
                return
        
        self.entries.append({
            'id': len(self.entries) + 1,
            'situation': situation,
            'lesson': lesson,
            'category': category,
            'created_at': datetime.now().isoformat(),
            'last_used': None,
            'use_count': 0,
            'reinforced': 0
        })
    
    def _is_similar(self, text1: str, text2: str, threshold: float = 0.7) -> bool:
        """简单的相似度判断"""
        words1 = set(text1.lower().split())
        words2 = set(text2.lower().split())
        
        if not words1 or not words2:
            return False
        
        intersection = len(words1 & words2)
        union = len(words1 | words2)
        
        return intersection / union >= threshold
    
    def retrieve_relevant(self, context: str = None, 
                         limit: int = 5) -> List[Dict[str, Any]]:
        """
        检索相关知识
        
        Args:
            context: 当前上下文/场景
            limit: 返回数量
            
        Returns:
            最相关的知识条目列表
        """
        if not context:
            sorted_entries = sorted(self.entries, 
                                   key=lambda x: x.get('use_count', 0),
                                   reverse=True)
            return sorted_entries[:limit]
        
        scored = []
        for entry in self.entries:
            score = self._relevance_score(context, entry)
            scored.append((score, entry))
        
        scored.sort(key=lambda x: x[0], reverse=True)
        result = [entry for _, entry in scored[:limit]]
        
        for entry in result:
            entry['last_used'] = datetime.now().isoformat()
            entry['use_count'] = entry.get('use_count', 0) + 1
        
        return result
    
    def _relevance_score(self, context: str, entry: Dict[str, Any]) -> float:
        """计算相关性分数"""
        score = 0.0
        
        context_lower = context.lower()
        situation = entry.get('situation', '').lower()
        lesson = entry.get('lesson', '').lower()
        
        keywords = ['dig', 'mine', 'attack', 'craft', 'smelt', 'explore',
                   'diamond', 'iron', 'nether', 'end', 'dragon', 'danger']
        
        for kw in keywords:
            if kw in context_lower:
                if kw in situation or kw in lesson:
                    score += 2.0
        
        if entry.get('last_used'):
            days_since = (datetime.now() - 
                         datetime.fromisoformat(entry['last_used'])).days
            if days_since < 7:
                score += 1.0
        
        score += entry.get('reinforced', 0) * 0.5
        
        return score
    
    def get_all_by_category(self, category: str) -> List[Dict]:
        return [e for e in self.entries if e.get('category') == category]
    
    def get_failure_lessons(self) -> List[Dict]:
        return self.get_all_by_category('failure')
    
    def search(self, query: str) -> List[Dict]:
        query_lower = query.lower()
        return [e for e in self.entries
                if query_lower in e.get('situation', '').lower()
                or query_lower in e.get('lesson', '').lower()]
```

## 六、性能优化

### 6.1 异步保存

```python
import threading
import time

class AsyncMemory(Memory):
    def __init__(self, *args, save_interval: int = 60, **kwargs):
        super().__init__(*args, **kwargs)
        self.save_interval = save_interval
        self._dirty = False
        self._save_thread = threading.Thread(target=self._save_loop, daemon=True)
        self._save_thread.start()
    
    def mark_dirty(self):
        self._dirty = True
    
    def _save_loop(self):
        while True:
            time.sleep(self.save_interval)
            if self._dirty:
                self.save()
                self._dirty = False
```
