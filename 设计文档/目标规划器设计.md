## 一、模块定位与目标

### 1.1 为什么需要规划器

当前的 Bot 只能**被动响应玩家指令**，没有自己的目标体系。没有规划器意味着：

| 问题 | 表现 | 影响 |
|------|------|------|
| 无目标感 | 玩家不说做什么就闲着 | 无法自主游玩 |
| 重复劳动 | 玩家让挖矿就一直挖 | 不知道何时该停止 |
| 进度混乱 | 玩家说"去杀末影龙" | 不会先检查装备和资源 | 
| 效率低下 | 不知道下一步最该做什么 | 做很多无用功 |

### 1.2 规划器的设计目标

```
核心功能：根据当前状态和最终目标，自动分解出最近的子目标
辅助功能：判断目标是否可执行，提示所需资源
```

### 2.1 阶段划分

| 阶段 | 名称 | 主要任务 | 关键里程碑 |
|------|------|----------|-----------|
| 0 | 生存准备 | 砍树、造工作台、熔炉 | 木质工具、熔炉 |
| 1 | 早期发展 | 采集石头、挖煤、铁器时代 | 石质工具、铁镐、铁护甲 |
| 2 | 中期积累 | 钻石、红石、附魔 | 钻石工具、附魔台 |
| 3 | 下界探索 | 进下界、建传送门、收集材料 | 黑曜石、下界要塞 |
| 4 | 末地准备 | 烈焰粉、末影之眼 | 激活末地传送门 |
| 5 | 击杀末影龙 | 准备装备、进入末地、战斗 | 击杀末影龙 |

建造传送门 ──────────────────────────→ 收集烈焰粉/珍珠 → 制作末影之眼
    ↓
进入下界 ─→ 收集材料 ─→ 激活传送门 ─→ 击杀末影龙（通关）
```

def get_goal(goal_id: str) -> Optional[Goal]:
    return GOALS.get(goal_id)


def get_available_goals(completed: Set[str]) -> List[Goal]:
    """获取当前可完成的目标（前置已满足）"""
    available = []
    for goal in GOALS.values():
        if goal.id in completed:
            continue
        if all(req in completed for req in goal.requires):
            available.append(goal)
    return sorted(available, key=lambda g: (-g.phase.value, -g.priority))


def get_next_goal(completed: Set[str]) -> Optional[Goal]:
    """获取下一个应该完成的目标"""
    available = get_available_goals(completed)
    return available[0] if available else None

"""
ai/planner/goals.py
目标管理
"""

from typing import List, Optional, Set, Dict
from enum import Enum
from dataclasses import dataclass

from .tech_tree import GOALS, Goal, get_goal, get_available_goals


class GoalStatus(Enum):
    NOT_STARTED = 'not_started'
    IN_PROGRESS = 'in_progress'
    COMPLETED = 'completed'
    FAILED = 'failed'


@dataclass
class GoalInstance:
    goal: Goal
    status: GoalStatus = GoalStatus.NOT_STARTED
    progress: float = 0.0
    attempts: int = 0
    failed_reason: str = None


class GoalQueue:
    """
    目标队列管理器
    """
    
    def __init__(self):
        self.completed: Set[str] = set()
        self.current: Optional[GoalInstance] = None
        self.queue: List[GoalInstance] = []
        self.goal_instances: Dict[str, GoalInstance] = {}
    
    def initialize_from_state(self, completed_goal_ids: List[str]):
        self.completed = set(completed_goal_ids)
        self._refresh_queue()
    
    def _refresh_queue(self):
        available = get_available_goals(self.completed)
        self.queue = []
        for goal in available:
            instance = GoalInstance(goal=goal)
            self.goal_instances[goal.id] = instance
            self.queue.append(instance)
        self.queue.sort(key=lambda g: (-g.goal.phase.value, -g.goal.priority))
    
    def get_current_goal(self) -> Optional[GoalInstance]:
        if self.current and self.current.status == GoalStatus.IN_PROGRESS:
            return self.current
        for instance in self.queue:
            if instance.status == GoalStatus.NOT_STARTED:
                return instance
        return None
    
    def start_goal(self, goal_id: str):
        instance = self.goal_instances.get(goal_id)
        if instance:
            instance.status = GoalStatus.IN_PROGRESS
            self.current = instance
    
    def complete_goal(self, goal_id: str):
        instance = self.goal_instances.get(goal_id)
        if instance:
            instance.status = GoalStatus.COMPLETED
            instance.progress = 1.0
            self.completed.add(goal_id)
            self.current = None
            self._refresh_queue()
    
    def fail_goal(self, goal_id: str, reason: str):
        instance = self.goal_instances.get(goal_id)
        if instance:
            instance.status = GoalStatus.FAILED
            instance.failed_reason = reason
            instance.attempts += 1
    
    def get_goal_path(self) -> List[str]:
        path = list(self.completed)
        for instance in self.queue:
            path.append(instance.goal.id)
        return path
    
    def get_formatted_path(self) -> str:
        path = self.get_goal_path()
        phase_names = ['生存', '早期', '中期', '下界', '末地准备', '终极']
        lines = []
        for goal_id in path:
            goal = get_goal(goal_id)
            status = "✅" if goal_id in self.completed else "🔄"
            lines.append(f"{status} {phase_names[goal.phase.value]}: {goal.name}")
        return "\n".join(lines)

"""
ai/planner/state_analyzer.py
状态分析
"""

from typing import Dict, List, Optional
from dataclasses import dataclass

from .tech_tree import GOALS, Goal


@dataclass
class FeasibilityReport:
    can_start: bool
    missing_resources: List[str]
    missing_abilities: List[str]
    suggestions: List[str]
    estimated_difficulty: str


class StateAnalyzer:
    """
    状态分析器
    """
    
    RESOURCE_REQUIREMENTS = {
        'collect_wood': {'log': 0},
        'collect_stone': {'wooden_pickaxe': 1},
        'collect_iron': {'stone_pickaxe': 1, 'furnace': 1},
        'collect_diamond': {'iron_pickaxe': 1},
        'collect_obsidian': {'diamond_pickaxe': 1},
    }
    
    def analyze_goal(self, goal: Goal, 
                    current_resources: Dict[str, int],
                    current_abilities: List[str]) -> FeasibilityReport:
        missing_resources = []
        missing_abilities = []
        suggestions = []
        
        # 检查前置目标
        for req_id in goal.requires:
            req_goal = GOALS.get(req_id)
            if req_goal:
                for resource, amount in req_goal.provides.items():
                    current = current_resources.get(resource, 0)
                    if current < amount:
                        missing_resources.append(f"{resource}: 需要 {amount}，当前 {current}")
        
        # 检查特定资源需求
        specific_req = self.RESOURCE_REQUIREMENTS.get(goal.id, {})
        for resource, needed in specific_req.items():
            current = current_resources.get(resource, 0)
            if current < needed:
                missing_resources.append(f"{resource}: 需要 {needed}，当前 {current}")
        
        # 检查缺失能力
        for ability in goal.grants_abilities:
            if ability not in current_abilities:
                missing_abilities.append(ability)
        
        can_start = len(missing_resources) == 0 and len(missing_abilities) == 0
        
        difficulty_map = {0: 'easy', 1: 'easy', 2: 'medium', 3: 'hard', 4: 'hard', 5: 'hard'}
        difficulty = difficulty_map.get(goal.phase.value, 'medium')
        
        return FeasibilityReport(
            can_start=can_start,
            missing_resources=missing_resources,
            missing_abilities=missing_abilities,
            suggestions=suggestions,
            estimated_difficulty=difficulty
        )

"""
ai/planner/planner.py
规划器主类
"""

from typing import Optional, Dict, List
import json
from pathlib import Path
import datetime

from .tech_tree import GOALS, Goal, get_goal
from .goals import GoalQueue, GoalInstance, GoalStatus
from .state_analyzer import StateAnalyzer, FeasibilityReport


class Planner:
    """
    Minecraft 智能体规划器
    """
    
    def __init__(self, state_file: str = '../memory/planner_state.json'):
        self.state_file = Path(state_file)
        self.queue = GoalQueue()
        self.analyzer = StateAnalyzer()
        self._load_state()
    
    def _load_state(self):
        if self.state_file.exists():
            try:
                with open(self.state_file, 'r', encoding='utf-8') as f:
                    data = json.load(f)
                    completed = data.get('completed_goals', [])
                    self.queue.initialize_from_state(completed)
                    print(f"📋 已加载 {len(completed)} 个已完成目标")
            except Exception as e:
                print(f"⚠️ 加载规划器状态失败: {e}")
                self.queue.initialize_from_state([])
        else:
            self.queue.initialize_from_state([])
    
    def _save_state(self):
        try:
            self.state_file.parent.mkdir(parents=True, exist_ok=True)
            with open(self.state_file, 'w', encoding='utf-8') as f:
                json.dump({
                    'completed_goals': list(self.queue.completed),
                    'saved_at': datetime.datetime.now().isoformat()
                }, f, indent=2)
        except Exception as e:
            print(f"❌ 保存规划器状态失败: {e}")
    
    def get_current_goal(self) -> Optional[GoalInstance]:
        return self.queue.get_current_goal()
    
    def analyze_current_goal(self, 
                            current_resources: Dict[str, int],
                            current_abilities: List[str]) -> FeasibilityReport:
        current = self.get_current_goal()
        if not current:
            return FeasibilityReport(
                can_start=False,
                missing_resources=[],
                missing_abilities=[],
                suggestions=['所有目标已完成！'],
                estimated_difficulty='easy'
            )
        
        return self.analyzer.analyze_goal(
            current.goal,
            current_resources,
            current_abilities
        )
    
    def complete_current_goal(self):
        current = self.get_current_goal()
        if current:
            self.queue.complete_goal(current.goal.id)
            self._save_state()
            print(f"✅ 目标完成: {current.goal.name}")
    
    def fail_current_goal(self, reason: str):
        current = self.get_current_goal()
        if current:
            self.queue.fail_goal(current.goal.id, reason)
            self._save_state()
    
    def get_next_action_hint(self, 
                            current_resources: Dict[str, int],
                            current_abilities: List[str]) -> Optional[str]:
        current = self.get_current_goal()
        if not current:
            return "🎉 所有目标已完成！"
        
        report = self.analyzer.analyze_goal(
            current.goal,
            current_resources,
            current_abilities
        )
        
        if report.can_start:
            goal = current.goal
            hints = {
                'collect_wood': "去砍树！收集至少 20 个木头。",
                'collect_stone': "用石镐挖石头，收集 30 个圆石。",
                'collect_iron': "挖铁矿，收集至少 24 个铁锭。",
                'collect_diamond': "去 Y=11-12 层挖钻石！",
                'collect_obsidian': "用钻石镐挖黑曜石，至少 10 个。",
                'nether_portal': "用黑曜石建造下界传送门（4x5），然后点燃。",
                'collect_blaze_rod': "在下界堡垒击杀烈焰人，收集烈焰棒。",
                'collect_ender_pearl': "击杀末影人收集珍珠。",
                'kill_dragon': "准备好装备，击杀末影龙！",
            }
            return hints.get(goal.id, f"执行目标: {goal.name}")
        
        else:
            if report.missing_resources:
                missing = ', '.join(report.missing_resources[:2])
                return f"⏳ 缺少资源: {missing}"
            return "⏳ 等待前置目标完成..."
    
    def get_goal_context_for_llm(self) -> str:
        parts = []
        current = self.get_current_goal()
        if current:
            parts.append(f"🎯 当前目标: {current.goal.name}")
            parts.append(f"   {current.goal.description}")
        
        path = self.queue.get_goal_path()
        if len(path) > 1:
            parts.append(f"📋 目标进度: {len(self.queue.completed)}/{len(path)} 完成")
        
        return "\n".join(parts)
    
    def reset(self):
        self.queue = GoalQueue()
        self._save_state()
        print("🔄 规划器已重置")
```

    current_abilities = []
